/* ====== Buttons behavior: copy + open (confirm) ====== */
document.querySelectorAll('.btn[data-href]').forEach(b=>{
  b.addEventListener('click', async (e)=>{
    const url = b.dataset.href;
    // try to open in new tab after confirm; also copy to clipboard
    try {
      const openConfirm = confirm('فتح الرابط في تبويب جديد؟ اضغط إلغاء للنسخ فقط');
      if(openConfirm){
        window.open(url, '_blank', 'noopener');
      } else {
        await navigator.clipboard.writeText(url);
        alert('تم نسخ الرابط');
      }
    } catch(err){
      // fallback: copy
      try{ await navigator.clipboard.writeText(url); alert('تم نسخ الرابط'); } catch(e){ alert(url); }
    }
  });

  // on long press (mobile) copy directly
  let pressTimer;
  b.addEventListener('touchstart', (ev)=> {
    pressTimer = setTimeout(async ()=> { try{ await navigator.clipboard.writeText(b.dataset.href); alert('تم نسخ الرابط'); }catch(e){} }, 700);
  });
  b.addEventListener('touchend', ()=> clearTimeout(pressTimer));
});

/* ====== Slow gentle starfield (canvas) ====== */
(() => {
  const canvas = document.getElementById('starfield');
  const ctx = canvas.getContext('2d');
  let w = canvas.width = innerWidth;
  let h = canvas.height = innerHeight;
  const num = Math.max(40, Math.floor((w*h)/8000)); // number of stars scaled
  const stars = [];

  function rand(min,max){ return Math.random()*(max-min)+min; }
  for(let i=0;i<num;i++){
    stars.push({
      x: Math.random()*w,
      y: Math.random()*h,
      r: rand(0.6,2.2),
      alpha: rand(0.06,0.25),
      vx: rand(-0.02,0.02),
      vy: rand(-0.02,0.02),
      phase: Math.random()*Math.PI*2
    });
  }

  function draw(){
    ctx.clearRect(0,0,w,h);
    // subtle gradient overlay to tint stars blue a bit
    for(const s of stars){
      s.x += s.vx;
      s.y += s.vy;
      s.phase += 0.002;
      // wrap
      if(s.x < -10) s.x = w+10;
      if(s.x > w+10) s.x = -10;
      if(s.y < -10) s.y = h+10;
      if(s.y > h+10) s.y = -10;

      const a = s.alpha + Math.sin(s.phase)*0.04;
      const grad = ctx.createRadialGradient(s.x, s.y, 0, s.x, s.y, s.r*6);
      grad.addColorStop(0, `rgba(80,160,255,${a})`);
      grad.addColorStop(0.4, `rgba(40,90,200,${a*0.45})`);
      grad.addColorStop(1, `rgba(15,20,40,0)`);
      ctx.beginPath();
      ctx.fillStyle = grad;
      ctx.arc(s.x, s.y, s.r*4, 0, Math.PI*2);
      ctx.fill();
    }

    requestAnimationFrame(draw);
  }
  draw();

  window.addEventListener('resize', ()=>{
    w = canvas.width = innerWidth;
    h = canvas.height = innerHeight;
    // scale stars array length lightly if resized drastically (keep simple)
  });

})();
